29/*********************************************************************
30
31 The ViewModel class implements a camera model, minus the perspective
32 transformation. It maintains the coordinate system of the camera
33 as three vectors (forward, up, and side), which are themselves
34 derived from two points (eye and lookat).
35
36 Apart from simplifying camera translation, this class provides
37 three rotation methods: yaw (rotation about the up axis),
38 roll (about the forward axis), and pitch (about the side axis).
39 Also, these rotations can take place about the eye (in which
40 case the eye location is fixed but the lookat point changes -
41 like rotating your head), or about the lookat point (in which
42 case your eye rotates around the point of interest).
43
44 There is also a routine for rotating the eye about an arbitrary
45 axis, which is quite useful in conjuction with the world up axis.
46
47 This class is heavily-dependent on the vec3 class in
48 the file algebra3.h.
49
50 The function update() sets the side and forward vectors based
51 on the lookat, eye, and up vectors. Remember to call this
52 function if you change the side or forward directly.
53
54 Sample use:
55 ViewModel vm;
56 vm.eye.set( 0.0, 0.0, 20.0 );
57 vm.lookat.set( 0.0, 0.0, 0.0 );
58 vm.update(); // updates the three vectors
59
60 vm.move( 0.0, 2.0, 0.0 ); // Moves the eye and lookat
61 // by 2 units in the world
62 // Y direction
63 vm.roll( 5.0 ); // rolls 5 degrees about the forward axis
64 vm.lookat_yaw( -25.0 ); // Yaws about the eye (lookat is
65 // fixed) by -25 degrees
66 vm.load_to_openGL(); // Sets OpenGL modelview matrix
67
68 .. render ...
69
70
71 ---------- List of member functions -----------------------------
72 set_distance() - Sets the distance from the eye to the lookat
73 set_up() - Sets the current up vector
74 set_eye() - Sets the current eye point
75 set_lookat() - Sets the current lookat point
76 roll() - Rolls about forward axis
77 eye_yaw() - Rotates eye about up vector
78 eye_yaw_abs() - Rotates eye about arbitrary axis
79 eye_pitch() - Rotates eye about side vector
80 lookat_yaw() - Rotates lookat about up vector
81 lookat_pitch() - Rotates lookat about side vector
82 reset_up_level() - Resets the up vector (after unwanted rolls), and
83 sets the eye level with the lookat point
84 move() - Moves eye and lookat by some amount
85 move_by_eye() - Moves eye to new position, lookat follows
86 move_by_lookat() - Moves lookat to new position, eye follows
87 move_abs() - Moves eye and lookat in world coordinates
88 rot_about_eye() - Rotates about the eye point by a given 4x4 matrix
89 rot_about_lookat() - Rotates about the lookat point by a given 4x4 matrix
90 make_mtx() - Constructs 4x4 matrix, used by load_to_openGL()
91 load_to_openGL() - Loads current camera description in openGL
92 load_to_openGL_noident() - Loads camera into OpenGL without first
93 resetting the OpenGL matrix to identity
94 reset() - Resets class values
95 ViewModel() - constructor
96 update() - Recalculates side and forward based on eye,
97 lookat, and up
98 dump() - Prints class contents to a file
99

111class ViewModel
112{
113public:
114 vec3 eye, lookat;
115 vec3 up, side, forward;
116 mat4 mtx;
117 float distance;
118
119 /******************************* set_distance() ***********/
120 /* This readjusts the distance from the eye to the lookat */
121 /* (changing the eye point in the process) */
122 /* The lookat point is unaffected */
123 void set_distance(float new_distance);
124
125 /******************************* set_up() ***************/
126 void set_up(const vec3 &new_up);
127
128 void set_up(float x, float y, float z);
129
130 /******************************* set_eye() ***************/
131 void set_eye(const vec3 &new_eye );
132
133 void set_eye(float x, float y, float z);
134
135 /******************************* set_lookat() ***************/
136 void set_lookat(const vec3 &new_lookat);
137
138 void set_lookat(float x, float y, float z);
139
140 /******************************* roll() *****************/
141 /* Rotates about the forward vector */
142 /* eye and lookat remain unchanged */
143 void roll(float angle);
144
145 /******************************* eye_yaw() *********************/
146 /* Rotates the eye about the lookat point, using the up vector */
147 /* Lookat is unaffected */
148 void eye_yaw(float angle);
149
150 /******************************* eye_yaw_abs() ******************/
151 /* Rotates the eye about the lookat point, with a specific axis */
152 /* Lookat is unaffected */
153 void eye_yaw_abs(float angle, const vec3 &axis);
154
155
156 /******************************* eye_pitch() ************/
157 /* Rotates the eye about the side vector */
158 /* Lookat is unaffected */
159 void eye_pitch(float angle);
160
161 /******************************* lookat_yaw()************/
162 /* This assumes the up vector is correct. */
163 /* Rotates the lookat about the side vector */
164 /* Eye point is unaffected */
165 void lookat_yaw(float angle);
166
167 /******************************* lookat_pitch() *********/
168 /* Rotates the lookat point about the side vector */
169 /* This assumes the side vector is correct. */
170 /* Eye point is unaffected */
171 void lookat_pitch(float angle);
172
173 /******************************* reset_up() ******************/
174 /* Resets the up vector to a specified axis (0=X, 1=Y, 2=Z) */
175 /* Also sets the eye point level with the lookat point, */
176 /* along the specified axis */
177 void reset_up(int axis_num);
178
179 void reset_up();
180
181 /******************************* move() ********************/
182 /* Moves a specified distance in the forward, side, and up */
183 /* directions. This function does NOT move by world */
184 /* coordinates. To move by world coords, use the move_abs */
185 /* function. */
186 void move(float side_move, float up_move, float forw_move);
187
188 void move(const vec3 &v);
189
190 /******************************* move_by_eye() ***********/
191 /* Sets the eye point, AND moves the lookat point by the */
192 /* same amount as the eye is moved. */
193 void move_by_eye(const vec3 &new_eye);
194
195 /******************************* move_by_lookat() *********/
196 /* Sets the lookat point, AND moves the eye point by the */
197 /* same amount as the lookat is moved. */
198 void move_by_lookat(const vec3 &new_lookat);
199
200 /******************************* move_abs() *****************/
201 /* Move the eye and lookat in world coordinates */
202 void move_abs(const vec3 &v);
203
204 /****************************** rot_about_eye() ************/
205 /* Rotates the lookat point about the eye, based on a 4x4 */
206 /* (pure) rotation matrix */
207 void rot_about_eye(const mat4 &rot);
208
209 /****************************** rot_about_lookat() ************/
210 /* Rotates the lookat point about the lookat, based on a 4x4 */
211 /* (pure) rotation matrix */
212 void rot_about_lookat(const mat4 &rot);
213
214 /******************************* make_mtx() *************/
215 /* Constructs a 4x4 matrix - used by load_to_openGL() */
216 void make_mtx();
217
218 /******************************* load_to_openGL() ********/
219 /* Sets the OpenGL modelview matrix based on the current */
220 /* camera coordinates */
221 void load_to_openGL();
222
223 /******************************* load_to_openGL_noident() ******/
224 /* Multiplies the current camera matrix by the existing openGL */
225 /* modelview matrix. This is same as above function, but */
226 /* does not set the OpenGL matrix to identity first */
227 void load_to_openGL_noident();
228
229 /******************************* reset() ****************/
230 /* Resets the parameters of this class */
231 void reset();
232
233 /******************************* ViewModel() ************/
234 /* Constructor */
235 ViewModel();
236
237 /******************************* update() ****************/
238 /* updates the view params. Call this after making */
239 /* direct changes to the vectors or points of this class */
240 void update();
241
242 /******************************* dump() *******************/
243 /* Prints the contents of this class to a file, typically */
244 /* stdin or stderr */
245 void dump(FILE *output) const;
246};
